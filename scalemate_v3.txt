<script>

const scales=[20,25,35,50,75,100,200,500];

function update(){

    const value=parseFloat(document.getElementById("real").value);

    if(isNaN(value)){
        table.innerHTML="";
        return;
    }

    let html="";

    scales.forEach(scale=>{

        let d=value/scale;

        html+=`
        <tr>
            <td>1:${scale}</td>
            <td>${d.toFixed(2).replace(/\.00$/,"")} мм</td>
        </tr>
        `;

    });

    table.innerHTML=html;

    resizeCanvas();

}

const canvas=document.getElementById("canvas");
const ctx=canvas.getContext("2d");

function resizeCanvas(){

    const panelHeight=document.getElementById("panel").offsetHeight;

    canvas.style.top=panelHeight+"px";

    canvas.width=window.innerWidth;

    canvas.height=window.innerHeight-panelHeight;

    ctx.lineWidth=2;
    ctx.lineCap="round";
    ctx.lineJoin="round";
    ctx.strokeStyle="rgba(0,0,0,0.75)";
}

window.addEventListener("resize",resizeCanvas);

resizeCanvas();

let drawing=false;

function pos(e){

    const r=canvas.getBoundingClientRect();

    if(e.touches){

        return{
            x:e.touches[0].clientX-r.left,
            y:e.touches[0].clientY-r.top
        };

    }

    return{
        x:e.clientX-r.left,
        y:e.clientY-r.top
    };

}

function start(e){

    drawing=true;

    const p=pos(e);

    ctx.beginPath();
    ctx.moveTo(p.x,p.y);

}

function draw(e){

    if(!drawing) return;

    e.preventDefault();

    const p=pos(e);

    ctx.lineTo(p.x,p.y);
    ctx.stroke();

}

function stop(){

    drawing=false;

}

canvas.addEventListener("mousedown",start);
canvas.addEventListener("mousemove",draw);
window.addEventListener("mouseup",stop);

canvas.addEventListener("touchstart",start,{passive:false});
canvas.addEventListener("touchmove",draw,{passive:false});
window.addEventListener("touchend",stop);

canvas.addEventListener("dblclick",()=>{

    ctx.clearRect(0,0,canvas.width,canvas.height);

});

</script>
